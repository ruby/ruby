Returns an array of the characters in +self+:

  'hello'.chars     # => ["h", "e", "l", "l", "o"]
  'тест'.chars      # => ["т", "е", "с", "т"]
  'こんにちは'.chars # => ["こ", "ん", "に", "ち", "は"]
  ''.chars          # => []
